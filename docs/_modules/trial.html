        self.mapWidget = MapWidget()

        self.button = QPushButton('Get Drawn')
        self.button.clicked.connect(self.buttonPushed)
        self.layout.addWidget(self.mapWidget)
        self.layout.addWidget(self.button)
        self.setLayout(self.layout)
        self.map = L.map(self.mapWidget, {'center': [12.97, 77.59], 'zoom': 10})
        self.m = L.marker([12.9342904,77.6097575])
        self.m.bindPopup('Home away from home?')
        self.map.addLayer(self.m)
        L.tileLayer('http://{s}.tile.osm.org/{z}/{x}/{y}.png').addTo(self.map)
        self.drawControl = L.control.draw()
        self.map.addControl(self.drawControl)
        self.show()
